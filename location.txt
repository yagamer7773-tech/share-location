<!doctype html>
<html>
<head><meta charset="utf-8"><title>Share location</title></head>
<body>
  <h2>Share your location</h2>
  <p id="status">Click the button to let this page get your location.</p>
  <button id="get">Get my location</button>
  <div id="out"></div>

<script>
const status = document.getElementById('status');
const out = document.getElementById('out');
document.getElementById('get').onclick = () => {
  if (!navigator.geolocation) {
    status.textContent = 'Geolocation not supported.';
    return;
  }
  status.textContent = 'Requesting locationâ€¦';
  navigator.geolocation.getCurrentPosition(pos => {
    const lat = pos.coords.latitude.toFixed(6);
    const lon = pos.coords.longitude.toFixed(6);
    status.textContent = 'Got location.';
    const gmap = `https://www.google.com/maps?q=${lat},${lon}`;
    out.innerHTML = `
      <p>Latitude: ${lat}<br>Longitude: ${lon}</p>
      <p><a href="${gmap}" target="_blank">Open in Google Maps</a></p>
      <p>
        <a id="wa" href="https://wa.me/?text=${encodeURIComponent('My location: ' + gmap)}" target="_blank">Send via WhatsApp</a>
      </p>
      <p>
        <a id="sms" href="sms:?&body=${encodeURIComponent('My location: ' + gmap)}">Send via SMS</a>
      </p>
    `;
  }, err => {
    status.textContent = 'Permission denied or error: ' + err.message;
  }, {enableHighAccuracy:true, timeout:10000});
};
</script>
</body>
</html>
